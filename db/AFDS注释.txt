create databases AFDS;
create table mainboard( 主板表
    id varchar 字符型主键 UUID
    品牌名 brands
    产品名 name
    产品型号名 model
    内存插槽数 ramnum
    m2插槽数 ssdnum
    sata硬盘插槽数 satanum
    外形尺寸 size
    电源插口 powernum
    供电模式 powermodel
    CPU类型 cpumodel
    CPU插槽 cpunum
);

create table memorybar( 内存条表
    id varchar 字符型主键 UUID
    品牌名
    产品名
    产品型号名
    适用类型 type
    内存容量 capacity
    ddr
    速度 speed
    内存主频 mainhz
    传输标准 std
    颗粒品牌 partbrands
    CL延迟 cl
    内存校验 ramcheck
);

create table cpu( CPU表
    id varchar 字符型主键 UUID
    品牌名
    产品名
    产品型号名

    适用类型
    CPU系列 generration
    制作工艺 nm
    核心代号 namnum
    封装大小 size
    核心数量 corenum
    线程数 threadsnum
);

create table videocard( 显卡表
    id varchar 字符型主键 UUID
    品牌名
    产品名
    产品型号名

    显卡类型
    显卡芯片 core
    核心频率 corehz
    显存频率 memhz
    显存容量 memnum
    显存位宽 weight
    电源接口 power
    供电模式 powertype
);

create table harddisk( 硬盘表
    id varchar 字符型主键 UUID
    品牌名
    产品名
    产品型号名

    type 硬盘类型 固态/普通
    存储容量
    接口类型 intertype
    硬盘尺寸
    主控芯片   core
    速度
);

create table power( 电源表
    id varchar 字符型主键 UUID
    品牌名
    产品名
    产品型号名
    
    电源类型 
    电源版本 version
    出线类型 outtype
    额定功率 stdw
    最大功率 maxw
    风扇描述 fan
    电源尺寸
);
